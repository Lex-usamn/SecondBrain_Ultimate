Python • Uso Diário Copiar Código Completo 
#!/usr/bin/env python3
"""
Exemplo de uso completo do Second Brain Engine
Este script demonstra o fluxo diário típico
"""

import sys
import json
from datetime import datetime

# Importar componentes do engine
from integrations.lex_flow_definitive import LexFlowClient, LexFlowConfig
from engine.core_engine import SecondBrainEngine
from engine.memory_system import MemorySystem


def main():
    """Fluxo principal do Segundo Cérebro"""
    
    # ============================================
    # 1. INICIALIZAÇÃO
    # ============================================
    
    "\n" + "="*60
    "🧠 SEGUNDO CÉREBRO INICIALIZANDO..."
    ("="*60 + "\n")
    
    # Configurar Lex Flow (já testado e funcionando!)
    lex_config = LexFlowConfig(
        username="Lex-Usamn",
        password="Lex#157.",
        base_url="https://flow.lex-usamn.com.br"
    )
    
    # Criar cliente Lex Flow
    lf = LexFlowClient(lex_config)
    
    # Inicializar Memory System (arquivos Obsidian)
    memory = MemorySystem(
        vault_path="./SecondBrain_Ultimate"
    )
    
    # Carregar identidade e perfil
    soul = memory.load_soul()     # Quem é o Second Brain
    user = memory.load_user()      Seu perfil operacional
    
    ✅ Identidade carregada:
    "   Propósito:", soul.get('purpose')[:60] + "...")
    "   Usuário:", user.get('username')
    
    # Inicializar Engine principal
    brain = SecondBrainEngine(
        lex_flow=lf,
        memory=memory,
        config={
            'debug': True,
            'auto_process_inbox': True
        }
    )
    
    "\n✅ SISTEMA PRONTO!\n")


    # ============================================
    # 2. BRIEFING MATINAL (O que fazer hoje?)
    # ============================================
    
    ("🌅 BRIEFING MATINAL".center(50, "-"))
    
    # Obter prioridades do Lex Flow
    priorities = lf.get_today_priorities()
    stats = lf.get_today_stats()
    
    # Enriquecer com IA (Decision Engine)
    morning_context = {
        'time_of_day': 'morning',
        'energy_level': user.get('morning_energy', 'medium'),
        'priorities_lex': priorities,
        'stats': stats,
        'goals_focus': ["Canais Dark", "Influencer"]
    }
    
    suggestion = brain.suggest_morning_plan(morning_context)
    
    ("\n🎯 TOP 3 PRIORIDADES DE HOJE:")
    for i, item in enumerate(suggestion['top_3'], 1):
        (f"   {i}. [{item['project']}] {item['task']}")
        (f"      ⏱️  Estimado: {item['estimated_time']} | 🎯 Porquê: {item['reason']}")
    
    ("\n💡 INSIGHT DO DIA:")
    (f"   {suggestion['daily_insight']}")


    # ============================================
    # 3. CAPTURA RÁPIDA (Durante o dia)
    # ============================================
    
    ("\n📥 CAPTURA DE IDEIA (Exemplo)".center(50, "-"))
    
    quick_capture(
        idea="Vídeo dark: Os 7 hospitais abandonados mais assustadores do Brasil",
        source='thought',
        context={'type': 'video_idea', 'niche': 'dark'}
    )
    
    ("\n✅ IDEIA CAPTURADA!")
        (f"   📍 ID: {captured['note_id']}")
        (f"   📁 Local: {captured['location']}")
        (f"   🏷️  Tags: {', '.join(captured['tags'])}")
        
        (f"   💡 Sugestão: Mover para '{captured['suggested_project']}'")
    

    # ============================================
    # 4. HEARTBEAT CHECK (Monitoramento)
    # ============================================
    
    ("\n💓 HEARTBEAT CHECK".center(50, "-"))
    
    heartbeat_report = brain.run_heartbeat_check()
    
    ("\n📊 STATUS DO SISTEMA:")
    (f"   Projetos ativos: {heartbeat_report['projects_active']}/{heartbeat_report['projects_total']}")
    (f"   Itens no Inbox: {heartbeat_report['inbox_size']}")
    (f"   Projetos parados: {len(heartbeat_report['stalled_projects'])}")
    
    ("\n⚠️  ALERTAS DETECTADOS:")
        for alert in heartbeat_report['alerts']:
            (f"   [{alert['level'].upper()}] {alert['message']}")
            (f"      → Ação: {alert['suggested_action']}")
    ("\n✅ Tudo OK! Nenhum alerta.")


    # ============================================
    # 5. PROCESSAMENTO DO INBOX (Fim do dia)
    # ============================================
    
    ("\n📥 PROCESSANDO INBOX".center(50, "-"))
    
    processing_result = brain.process_inbox_with_ai()
    
    ("\n📋 RESULTADO DO PROCESSAMENTO:")
    (f"   Notas processadas: {processing_result['processed_count']}")
    (f"   Movidas para projetos: {processing_result['moved_to_projects']}")
    (f"   Arquivadas: {processing_result['archived']}")
    (f"   Ainda pendentes: {processing_result['remaining']}")
    
    ("\n🤖 CATEGORIZAÇÃO IA:")
        for cat in processing_result['categorization']:
            (f"   • '{cat['item']}' → {cat['suggested_area']}")


    # ============================================
    # 6. ATUALIZAR MEMÓRIA (Learning)
    # ============================================
    
    ("\n💾 ATUALIZANDO MEMÓRIA...".center(50, "-"))
    
    add_lesson(
        category='Geral',
        lesson='Teste: Captura via thought funciona bem para ideias espontâneas',
        context='Captura rápida teste matinal'
    )
    
    update_metrics({
        'date': datetime.now().strftime('%Y-%m-%d'),
        'ideas_captured': 1,
        'inbox_processed': processing_result['processed_count'],
        'heartbeats_run': 1
    })
    
    ("✅ Memória atualizada!")


    # ============================================
    # 7. RESUMO FINAL
    # ============================================
    
    ("\n" + "="*60)
    ("📊 RESUMO DA SESSÃO".center(56))
    ("="*60)
    
    summary = brain.generate_session_summary()
    
    ("\n📈 Métricas da Sessão:")
    for key, value in summary.items():
        (f"   • {key}: {value}")
    
    ("\n🎯 Próximos Passos Sugeridos:")
    next_steps = brain.suggest_next_steps()
    for step in next_steps[:3]:
        (f"   → {step}")
    
    ("\n" + "="*60)
    ("🧠 SEGUNDO CÉREBRO ENCERRADO COM SUCESSO".center(54))
    ("="*60 + "\n")


if __name__ == "__main__":
    main()
